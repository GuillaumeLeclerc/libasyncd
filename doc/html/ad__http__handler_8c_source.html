426  // Parse request line.
427  char *saveptr;
428  char *method = strtok_r(line, " ", &saveptr);
429  char *uri = strtok_r(NULL, " ", &saveptr);
430  char *httpver = strtok_r(NULL, " ", &saveptr);
431  char *tmp = strtok_r(NULL, " ", &saveptr);
432 
433  if (method == NULL || uri == NULL || httpver == NULL || tmp != NULL) {
434  DEBUG("Invalid request line. %s", line);
435  return AD_HTTP_ERROR;
436  }
437 
438  // Set request method
439  http->request.method = qstrupper(strdup(method));
440 
441  // Set HTTP version
442  http->request.httpver = qstrupper(strdup(httpver));
443  if (strcmp(http->request.httpver, HTTP_PROTOCOL_09)
444  && strcmp(http->request.httpver, HTTP_PROTOCOL_10)
445  && strcmp(http->request.httpver, HTTP_PROTOCOL_11)
446  ) {
447  DEBUG("Unknown protocol: %s", http->request.httpver);
448  return AD_HTTP_ERROR;
449  }
450 
451  // Set URI
452  if (uri[0] == '/') {
453  http->request.uri = strdup(uri);
454  } else if ((tmp = strstr(uri, "://"))) {
455  // divide URI into host and path
456  char *path = strstr(tmp + CONST_STRLEN("://"), "/");
457  if (path == NULL) { // URI has no path ex) http://domain.com:80
458  http->request.headers->putstr(http->request.headers, "Host", tmp + CONST_STRLEN("://"));
459  http->request.uri = strdup("/");
460  } else { // URI has path, ex) http://domain.com:80/path
461  *path = '\0';
462  http->request.headers->putstr(http->request.headers, "Host", tmp + CONST_STRLEN("://"));
463  *path = '/';
464  http->request.uri = strdup(path);
465  }
466  } else {
467  DEBUG("Invalid URI format. %s", uri);
468  return AD_HTTP_ERROR;
469  }
470 
471  // Set request path. Only path part from URI.
472  http->request.path = strdup(http->request.uri);
473  tmp = strstr(http->request.path, "?");
474  if (tmp) {
475  *tmp ='\0';
476  http->request.query = strdup(tmp + 1);
477  } else {
478  http->request.query = strdup("");
479  }
480  qurl_decode(http->request.path);
481 
482  // check path
483  if (isValidPathname(http->request.path) == false) {
484  DEBUG("Invalid URI format : %s", http->request.uri);
485  return AD_HTTP_ERROR;
486  }
487  correctPathname(http->request.path);
488 
489  DEBUG("Method=%s, URI=%s, VER=%s", http->request.method, http->request.uri, http->request.httpver);
490 
491  return AD_HTTP_REQ_REQUESTLINE_DONE;
492 }

567  // Peek chunk size.
568  size_t crlf_len = 0;
569  char *line = evbuffer_peekln(in, &crlf_len, EVBUFFER_EOL_CRLF);
570  if (line == NULL) return -1; // not enough data.
571  size_t linelen = strlen(line);
572 
573  // Parse chunk size
574  int chunksize = -1;

576  free(line);
577  if (chunksize < 0) return -2; // format error
578 
579  // Check if we've received whole data of this chunk.
580  size_t datalen = linelen + crlf_len + chunksize + crlf_len;
581  size_t inbuflen = evbuffer_get_length(in);
582  if (inbuflen < datalen) {
583  return -1; // not enough data.
584  }
585 
586  // Copy chunk body
587  evbuffer_drainln(in, NULL, EVBUFFER_EOL_CRLF);
588  http_add_inbuf(in, http, chunksize);
589  evbuffer_drainln(in, NULL, EVBUFFER_EOL_CRLF);
590

592 }
